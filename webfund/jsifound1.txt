function sigma(num){
    var sum = 0;
    for(i=1;i<=num;i++)
        sum = sum +i;

    return sum;
}

console.log(sigma(3));
console.log(sigma(5));
console.log(sigma(10));

function factorial(num){
    var product = 1;
    for(i=1;i<=num;i++)
        product = product*i;

    return product;
}

console.log(factorial(3));
console.log(factorial(5));
console.log(factorial(10));
